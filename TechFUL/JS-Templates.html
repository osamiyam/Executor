Executorでプログラムを開発するときのテンプレート

以下のコードをExecutorにコピーして使ってください．また，入力は配列stdinとして定義してください．入力は関数kadaiではsとして与えられます．sは配列で，先頭のデータはs[0]となります．データが整数のときには関数intを通して利用してください．また，データが浮動小数点数の場合にはfloatを一度通して利用してください．文字列の場合にはそのまま使ってください．

// プログラムを試すために必要な入力 //
stdin = [4, 23, 22, 0, 21.4, "abc"]

//プログラム本体//
function kadai(s){
    var n = int(s[0])
    var sum = 0
    for (var i = 0; i < n; i++){
        sum += float(s[i + 1])
    }
    puts(sum / n)
}

//-------この部分を付け加える--------//
var int = parseInt
var float = parseFloat
kadai(stdin)
//-----------------------------//

TechFULシステムにプログラムを提出するときのテンプレート

作成したプログラムをTechFULに提出するときは以下のようなテンプレートを使ってください．作成したプログラムkadai はそのままコピーして，その後ろに「この部分を付け加える」の部分をつけてください．入力はTechFULから与えれるので必要ありません．

// プログラム本体 //
function kadai(s){
    var n = int(s[0])
    var sum = 0
    for (var i = 0; i < n; i++){
        sum += float(s[i + 1])
    }
    puts(sum / n)
}

//-------この部分を付け加える--------//
var puts = console.log
var int = parseInt
var float = parseFloat
process.stdin.resume();
process.stdin.setEncoding("utf8");
process.stdin.on("data", function(inputData) {
    kadai(inputData.toString().split('\n'));
});
//-------------------------------//
